    <example id="1" description="Получение клиентов с суммой заказов за квартал (PostgreSQL)">
        <input>
            Задача: Получить список клиентов с суммой всех заказов за последний квартал.
            СУБД: PostgreSQL
            Таблицы: customers (customer_id, full_name), orders (order_id, customer_id, total_amount, order_date)
            Связь: customers.customer_id → orders.customer_id
        </input>
        <output>
```sql
-- Объединение таблиц customers и orders, расчёт общей суммы заказов за последние 3 месяца
SELECT
    c.customer_id,
    c.full_name,
    COUNT(o.order_id) AS order_count,
    SUM(o.total_amount) AS total_spent
FROM
    customers c
JOIN
    orders o ON c.customer_id = o.customer_id
WHERE
    o.order_date >= CURRENT_DATE - INTERVAL '3 months'
GROUP BY
    c.customer_id, c.full_name
ORDER BY
    total_spent DESC;
```
        </output>
    </example>
